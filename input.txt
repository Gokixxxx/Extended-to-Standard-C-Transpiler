fn apply(f, x) {
    return f(x);
}

fn apply_closure(f, x) {
    return f(x);
}

let doubled = fn(y) => y * 2;
let r1 = apply(doubled, 5);

let make_adder = fn(a) => fn(b) => a + b;
let add5 = make_adder(5);
let r2 = apply_closure(add5, 10);

let add3 = make_adder(3);
let r3 = apply_closure(add3, 7);

fn doubled(x) { return x * 2; }

fn add(a) { let ret = fn(b) { return a + b; }; return ret; }
let f = add(3);
print(f(4));

struct Rectangle { width: i32, height: i32 }
impl Rectangle { fn area(&self) { return self.width * self.height; } }
let rect = new Rectangle { width: 5, height: 6 };
print(rect.area());

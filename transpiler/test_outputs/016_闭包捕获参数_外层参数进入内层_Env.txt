----------------------------------------------------------------------
【生成的 C 代码】

#include "option.h"
#include "vec.h"
#include "closure.h"

struct __env_1 {
    int base;
};

static int __fn_1(void *__env, int x) {
    struct __env_1 *env = (struct __env_1 *)__env;
    int base = env->base;
    return (base + x);
}

Closure_i32_i32 make_adder2(int base) {
    struct __env_1 *__t1 = malloc(sizeof(struct __env_1));
    __t1->base = base;
    Closure_i32_i32 __t2 = {__t1, __fn_1};
    return __t2;
}

int main() {
    Closure_i32_i32 add5 = make_adder2(5);
    Closure_i32_i32 add100 = make_adder2(100);
    int r8 = add5.fn(add5.env, 3);
    int r9 = add100.fn(add100.env, 3);
    free(add5.env);
    free(add100.env);
    return 0;
}

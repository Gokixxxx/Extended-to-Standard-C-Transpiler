测试 #34: Vec 与 struct 混用
======================================================================

----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
struct Rect {
            w: i32,
            h: i32
        }

        impl Rect {
            fn area(&self) {
                return self.w * self.h;
            }
        }

        let v = [1, 2, 3];
        let rect = new Rect { w: 2, h: 3 };
        let r7 = rect.area() + v[1];

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

typedef struct {
    int w;
    int h;
} Rect;

int Rect_area(Rect *self) {
    return ((self)->w * (self)->h);
}

int main() {
    
    Vec_i32 v = vec_new_i32();
    vec_push_i32(&v, 1);
    vec_push_i32(&v, 2);
    vec_push_i32(&v, 3);
    Rect rect = (Rect){2, 3};
    int r7 = (Rect_area(&(rect)) + vec_get_i32(v, 1));
    free(v.data);
    return 0;
}

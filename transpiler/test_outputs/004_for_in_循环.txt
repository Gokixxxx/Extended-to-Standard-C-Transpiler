【输入源码】

fn mytest() {
    let arr = [1, 2, 3];
    let sum = 0;
    for x in arr {
        sum = sum + x;
    }
}

    vec_push_i32(&arr, 2);
    vec_push_i32(&arr, 3);

        int __t1 = 0;

    free(arr.data);
}

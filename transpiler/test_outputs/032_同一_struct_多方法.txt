测试 #32: 同一 struct 多方法
======================================================================

----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
struct Counter {
            value: i32
        }

        impl Counter {
            fn get(&self) {
                return self.value;
            }

            fn add(&self, delta) {
                return self.value + delta;
            }
        }

        let c = new Counter { value: 100 };
        let r4 = c.get();
        let r5 = c.add(7);

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

typedef struct {
    int value;
} Counter;

int Counter_get(Counter *self) {
    return (self)->value;
}

int Counter_add(Counter *self, int delta) {
    return ((self)->value + delta);
}

int main() {
    
    Counter c = (Counter){100};
    int r4 = Counter_get(&(c));
    int r5 = Counter_add(&(c), 7);
    return 0;
}

struct Score {
            val: i32
        }

        impl Score {
            fn is_pass(&self) {
                if (self.val >= 60) {
                    return 1;
                } else {
                    return 0;
                }
            }
        }

        let s1 = new Score { val: 80 };
        let s2 = new Score { val: 50 };
        let r13 = s1.is_pass();
        let r14 = s2.is_pass();

#include "vec.h"
#include "closure.h"

typedef struct {
    int val;
} Score;

int Score_is_pass(Score *self) {
    if (((self)->val >= 60)) {
        return 1;
    }
    else {
        return 0;
    }
}

int main() {
    
    Score s1 = (Score){80};
    Score s2 = (Score){50};
    int r13 = Score_is_pass(&(s1));
    int r14 = Score_is_pass(&(s2));

----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
struct Box {
            width: i32,
            height: i32
        }

        impl Box {
            fn max_side(&self) {
                if (self.width > self.height) {
                    return Some(self.width);
                } else {
                    return Some(self.height);
                }
            }
        }

        let b = new Box { width: 8, height: 5 };
        let opt = b.max_side();
        let r8 = match opt {
            Some(v) => v,
            None => 0
        };

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

typedef struct {
    int width;
    int height;
} Box;

Option_i32 Box_max_side(Box *self) {
    if (((self)->width > (self)->height)) {
        return Some_i32((self)->width);
    }
    else {
        return Some_i32((self)->height);
    }
}

int main() {
    
    Box b = (Box){8, 5};
    Option_i32 opt = Box_max_side(&(b));
    int r8;
    if (opt.is_some) {
        r8 = opt.value;
    } else {
        r8 = 0;
    }
    return 0;
}

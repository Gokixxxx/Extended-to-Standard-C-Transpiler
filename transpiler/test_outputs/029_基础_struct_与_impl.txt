----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
struct Rectangle {
            width: i32,
            height: i32
        }

        impl Rectangle {
            fn area(&self) {
                return self.width * self.height;
            }
        }

        let rect = new Rectangle { width: 30, height: 50 };
        let r1 = rect.area();

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

typedef struct {
    int width;
    int height;
} Rectangle;

int Rectangle_area(Rectangle *self) {
    return ((self)->width * (self)->height);
}

int main() {
    
    Rectangle rect = (Rectangle){30, 50};
    int r1 = Rectangle_area(&(rect));
    return 0;
}

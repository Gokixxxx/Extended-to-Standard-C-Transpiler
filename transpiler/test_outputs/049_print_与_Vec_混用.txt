        print(v[1]);

#include "option.h"
#include "vec.h"
#include "closure.h"

int main() {
    Vec_i32 v = vec_new_i32();
    vec_push_i32(&v, 10);
    vec_push_i32(&v, 20);
    vec_push_i32(&v, 30);
    printf("%d\n", vec_get_i32(v, 1));
    int r10 = vec_get_i32(v, 2);
    free(v.data);
    return 0;
}

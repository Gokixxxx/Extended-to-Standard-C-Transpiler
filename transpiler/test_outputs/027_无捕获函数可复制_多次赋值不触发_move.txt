----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
let doubled = fn(x) => x * 2;
        let d1 = doubled;
        let d2 = doubled;
        let r4 = d1(3);
        let r5 = d2(4);

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

static int __fn_1(int x) {
    return (x * 2);
}
static int __fn_1_closure(void *__env, int x) {
    (void)__env;
    return __fn_1(x);
}

int main() {
    int (*doubled)(int) = __fn_1;
    int (*d1)(int) = doubled;
    int (*d2)(int) = doubled;
    int r4 = d1(3);
    int r5 = d2(4);
    return 0;
}

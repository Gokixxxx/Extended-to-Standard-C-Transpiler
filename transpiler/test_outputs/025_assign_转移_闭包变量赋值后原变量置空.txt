测试 #25: assign 转移：闭包变量赋值后原变量置空
======================================================================

----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
let b = 1;
        let f1 = fn(x) => b + x;
        let f2 = fn(x) => b * 2 + x;
        f2 = f1;
        let r2 = f2(10);

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

struct __env_1 {
    int b;
};

struct __env_2 {
    int b;
};

static int __fn_1(void *__env, int x) {
    struct __env_1 *env = (struct __env_1 *)__env;
    int b = env->b;
    return (b + x);
}

static int __fn_2(void *__env, int x) {
    struct __env_2 *env = (struct __env_2 *)__env;
    int b = env->b;
    return ((b * 2) + x);
}

int main() {
    int b = 1;
    struct __env_1 *f1_env = malloc(sizeof(struct __env_1));
    f1_env->b = b;
    Closure_i32_i32 f1 = {f1_env, __fn_1};
    struct __env_2 *f2_env = malloc(sizeof(struct __env_2));
    f2_env->b = b;
    Closure_i32_i32 f2 = {f2_env, __fn_2};
    free(f2.env);
    f2 = f1;
    f1.env = NULL;
    int r2 = f2.fn(f2.env, 10);
    free(f1.env);
    free(f2.env);
    return 0;
}

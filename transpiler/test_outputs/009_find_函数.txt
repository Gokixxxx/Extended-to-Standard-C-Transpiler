【输入源码】

fn find(arr, target) {
    for x in arr {
        if x > target {
            return Some(x);
        }
    }
    return None;
}

fn mytest() {
    let result1 = find([1, 3, 5, 2, 4], 3);
    let result2 = find([1, 2, 3], 5);
}

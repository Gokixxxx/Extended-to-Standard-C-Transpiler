#include "option.h"
#include "vec.h"
#include "closure.h"

struct __env_1 {
    int x;
};

static int __fn_1(void *__env, int y) {
    struct __env_1 *env = (struct __env_1 *)__env;
    int x = env->x;
    return (x + y);
}

Closure_i32_i32 make_adder(int x) {
    struct __env_1 *adder_env = malloc(sizeof(struct __env_1));
    adder_env->x = x;
    Closure_i32_i32 adder = {adder_env, __fn_1};
    return adder;
}

int main() {
    Closure_i32_i32 add5 = make_adder(5);
    int r3 = add5.fn(add5.env, 3);
    free(add5.env);
    return 0;
}

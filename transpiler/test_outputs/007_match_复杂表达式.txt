【输入源码】

fn mytest() {
    let x = Some(5);
    let y = match x {
        Some(v) => v * 2 + 1,
        None => 0
    };
}

#include "vec.h"
#include "closure.h"

void mytest(void) {
    Option_i32 x = Some_i32(5);
    int y;
    if (x.is_some) {
        y = ((x.value * 2) + 1);
    } else {
        y = 0;
    }
}

    return 0;

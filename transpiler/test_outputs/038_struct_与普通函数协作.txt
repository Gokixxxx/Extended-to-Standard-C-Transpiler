======================================================================

----------------------------------------------------------------------
【输入源码】
----------------------------------------------------------------------
struct Item {
            price: i32,
            qty: i32
        }

        impl Item {
            fn total(&self) {
                return self.price * self.qty;
            }
        }

        fn calc(item) {
            return item.total();
        }

        let i = new Item { price: 7, qty: 6 };
        let r12 = calc(i);

----------------------------------------------------------------------
【生成的 C 代码】
----------------------------------------------------------------------
#include "option.h"
#include "vec.h"
#include "closure.h"

typedef struct {
    int price;
    int qty;
} Item;

int Item_total(Item *self) {
    return ((self)->price * (self)->qty);
}

int calc(Item item) {
    return Item_total(&(item));
}

int main() {
    
    Item i = (Item){7, 6};
    int r12 = calc(i);
    return 0;
}
